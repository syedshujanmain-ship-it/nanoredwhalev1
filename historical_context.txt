          const putRes = await fetch(`https://api.github.com/repos/${owner}/${repoName}/contents/${encodedPath}`, {
            method: 'PUT',
            headers: authHeaders,
            body: JSON.stringify({

              content: btoaUnicode(file.content),
            }),
          });

            pushedCount++;
            continue;
          }

          // If 422, file may already exist — get its SHA and update
          if (putRes.status === 422 || putRes.status === 409) {
            const getRes = await fetch(`https://api.github.com/repos/${owner}/${repoName}/contents/${encodedPath}?ref=${branch}`, {

              const existingSha = Array.isArray(fileInfo) ? fileInfo[0]?.sha : fileInfo.sha;

                  method: 'PUT',
                  headers: authHeaders,
                  body: JSON.stringify({
                    message: `Update ${filePath} — RED WHALE V1`,
                    content: btoaUnicode(file.content),
                    sha: existingSha,
                  }),
                });
                if (updateRes.ok) {
                  pushedCount++;
                  continue;
                }
              }
            }
          }
          skippedCount++;
        } catch (e) {
          skippedCount++;
          console.error(`Failed to push ${filePath}:`, e);
        }
      }

      const repoUrl = `https://github.com/${owner}/${repoName}`;
      setLastPushedUrl(repoUrl);

      toast.success(`Pushed ${pushedCount}/${validFiles.length} files to GitHub`, {
        description: repoUrl,
        duration: 15000,
      });

      if (skippedCount > 0) {
        toast.warning(`${skippedCount} files could not be pushed`, { duration: 5000 });
      }
    } catch (error: any) {
      toast.error('GitHub push failed', { description: error.message, duration: 10000 });
      console.error('GitHub push error:', error);
    } finally {
      setPushingToGitHub(false);
    }

// Retry fetch with exponential backoff
async function fetchWithRetry(
  url: string,
  options: RequestInit,
  retries = 3,
  delay = 1000
): Promise {
  for (let i = 0; i < retries; i++) {
    const res = await fetch(url, options);
    // Retry on rate limit (403 with abuse rate limit) or server errors (5xx)
    if (res.status === 403 || res.status >= 500) {
      const errBody = await res.text();
      // Check if it's a rate limit
      if (errBody.includes('rate limit') || errBody.includes('abuse') || res.status >= 500) {
        if (i < retries - 1) {
          await new Promise(r => setTimeout(r, delay * Math.pow(2, i)));
          continue;
        }
      }
    }
    return res;
  }
  return new Response('', { status: 500 });
}
